003import android.arch.persistence.room.ColumnInfo;
004import android.arch.persistence.room.Entity;

009import java.util.Date;

011/**
012 * This is the entity invoice.class that lists all the fields, with the necessary setters and getters

014 */
015
016@Entity(

043  /**

050
051  /**
052   * This is the setter for id
053   * @param  id
054   */
055  public void setId(long id) {
056    this.id = id;
057  }
058
059  /**
060   * Thia is the getter for completed
061   * @return completed
062   */
063  public Date getCompleted() {
064    return completed;
065  }
066
067  /**
068   * Thia is the setter for completed
069   * @return completed
070   */
071  public void setCompleted(Date completed) {
072    this.completed = completed;

074
075  /**
076   * Thia is the getter for invoiced
077   * @return invoiced
078   */
079  @NonNull
080  public Date getInvoiced() {
081    return invoiced;
082  }
083  /**
084   * Thia is the getter for invoiced
085   * @return invoiced
086   */
087  public void setInvoiced(@NonNull Date invoiced) {
088    this.invoiced = invoiced;
089  }
090
091  /**
092   * Thia is the getter for price
093   * @return price
094   */
095  public long getPrice() {
096    return price;
097  }
098
099  /**
100   * Thia is the setter for price
101   * @return price
102   */
103  public void setPrice(long price) {
104    this.price = price;
105  }
106
107  /**
108   * Thia is the getter for JobId
109   * @return jobId
110   */
111  public long getJobId() {
112    return jobId;
113  }
114
115  /**
116   * Thia is the setter for jobId
117   * @return jobId
118   */
119  public void setJobId(long jobId) {
120    this.jobId = jobId;
121  }
122
123  /**
124   * Thia is the getter for address
125   * @return address
126   */
127  public String getAddress() {
128    return address;
129  }
130
131  /**
132   * Thia is the setter for address
133   * @return address
134   */
135  public void setAddress(String address) {
136    this.address = address;
137  }
138}
