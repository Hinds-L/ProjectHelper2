054    @Override
055    public boolean onNavigationItemSelected(@NonNull MenuItem item) {
056      boolean handled = true;
057      Fragment fragment = null;
058  switch (item.getItemId()) {
059        case R.id.navigation_quotes:
060          fragment = new QuotesFragment();
061          break;
062        case R.id.navigation_invoice:
063          fragment = new InvoicesFragment();
064          break;
065        case R.id.navigation_job:
066          fragment = new JobsFragment();
067          break;
068        default:
069          handled = false;
070
071      }
072      if (handled) {
073
074        //fragmentTitle.setText(item.getTitle());
075
076        if (fragment != null) {
077          getSupportFragmentManager().beginTransaction()
078              .replace(R.id.fragment_container, fragment)
079              .commit();
080
081        }
082
083      }
084      return handled;
085    }
