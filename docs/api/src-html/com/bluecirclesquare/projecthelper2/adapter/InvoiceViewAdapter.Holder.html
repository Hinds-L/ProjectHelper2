001package com.bluecirclesquare.projecthelper2.adapter;
002
003import static android.text.format.DateFormat.getDateFormat;
004
005import android.content.Context;
006import android.support.annotation.NonNull;
007import android.support.v7.widget.RecyclerView;
008import android.view.LayoutInflater;
009import android.view.View;
010import android.view.ViewGroup;
011import android.widget.TextView;
012import com.bluecirclesquare.projecthelper2.R;
013import com.bluecirclesquare.projecthelper2.model.entity.Invoice;
014import java.text.DateFormat;
015import java.util.List;
016
017/**
018 * This is the InvoiceViewAdapter.class that uses RecyclerView to populate list items
019 */
020
021public class InvoiceViewAdapter extends RecyclerView.Adapter<InvoiceViewAdapter.Holder> {
022
023
024  private final Context context;
025  private final List<Invoice> invoices;
026
027  /**
028   * Gets the list of invoices from the adapter
029   * @param context
030   * @param invoices
031   */
032  public InvoiceViewAdapter(Context context, List<Invoice> invoices) {
033    this.context = context;
034    this.invoices = invoices;
035  }
036
037  @NonNull
038  @Override
039  public Holder onCreateViewHolder(@NonNull ViewGroup parent, int viewType) {
040    View view = LayoutInflater.from(context).inflate(R.layout.invoice_list_item, parent, false);
041    return new Holder(view);
042
043  }
044
045  @Override
046  public void onBindViewHolder(@NonNull Holder holder, int position) {
047    holder.bind(invoices.get(position));
048  }
049
050  @Override
051  public int getItemCount() {
052    return invoices.size();
053  }
054
055  /**
056   * Holds the fields that will be populated by the recyclerview
057   */
058  public class Holder extends RecyclerView.ViewHolder {
059
060    private Invoice invoice;
061    private TextView id;
062    private TextView invoiced;
063    private TextView price;
064    private TextView address;
065
066    /**
067     * References the location of items
068     */
069
070    public Holder(@NonNull View itemView) {
071      super(itemView);
072      id = itemView.findViewById(R.id.invoice_id);
073      invoiced = itemView.findViewById(R.id.date_invoiced);
074      price = itemView.findViewById(R.id.price);
075      address = itemView.findViewById(R.id.job_address);
076    }
077
078    //  @Override
079    //   public void onCreateContextMenu(ContextMenu menu, View v, ContextMenuInfo menuInfo)}
080
081    /**
082     * Sets the text for the items listed in the invoice class
083     * @param invoice
084     */
085    private void bind(Invoice invoice) {
086      DateFormat format = getDateFormat(context);
087      this.invoice = invoice;
088      id.setText(Long.toString(invoice.getId()));
089      invoiced.setText(format.format(invoice.getInvoiced()));
090      price.setText(context.getString(R.string.amount_format, invoice.getPrice() / 100d));
091      address.setText(invoice.getAddress());
092    }
093  }
094}
