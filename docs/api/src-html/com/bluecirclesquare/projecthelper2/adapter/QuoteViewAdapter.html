001package com.bluecirclesquare.projecthelper2.adapter;
002
003import android.content.Context;
004import android.support.annotation.NonNull;
005import android.support.v7.widget.RecyclerView;
006import android.util.Log;
007import android.view.LayoutInflater;
008import android.view.View;
009import android.view.ViewGroup;
010import android.widget.TextView;
011import com.bluecirclesquare.projecthelper2.R;
012import com.bluecirclesquare.projecthelper2.model.entity.Quote;
013import java.util.List;
014
015public class QuoteViewAdapter extends RecyclerView.Adapter<QuoteViewAdapter.Holder> {
016
017  private final Context context;
018  private final List<Quote> quotes;
019
020  public QuoteViewAdapter(Context context, List<Quote> quotes) {
021    this.context = context;
022    this.quotes = quotes;
023  }
024
025  @NonNull
026  @Override
027  public Holder onCreateViewHolder(@NonNull ViewGroup parent, int viewType) {
028    View view = LayoutInflater.from(context).inflate(R.layout.quote_list_item, parent, false);
029    return new Holder(view);
030  }
031
032  @Override
033  public void onBindViewHolder(@NonNull Holder holder, int position) {
034    holder.bind(quotes.get(position));
035  }
036
037  @Override
038  public int getItemCount() {
039    Log.v("QuoteViewAdapter", "Count Called: "+quotes.size());
040    return quotes.size();
041  }
042
043  public class Holder extends RecyclerView.ViewHolder {
044
045    private Quote quote;
046    private TextView contact;
047    private TextView amount;
048    private TextView quoteNumber;
049
050    public Holder(@NonNull View itemView) {
051      super(itemView);
052      contact = itemView.findViewById(R.id.contact);
053      amount = itemView.findViewById(R.id.amount);
054      quoteNumber = itemView.findViewById(R.id.quote_number);
055    }
056
057//    @Override
058//    public void onCreateContextMenu(ContextMenu menu, View v, ContextMenuInfo menuInfo) {
059//
060//    }
061
062    private void bind(Quote quote) {
063      this.quote = quote;
064      contact.setText(quote.getContact());
065      quoteNumber.setText(quote.getQuoteNumber());
066      amount.setText(context.getString(R.string.amount_format, quote.getAmount() / 100d));
067    }
068  }
069
070}
